Code javascript, khởi tạo modal khi element có class "show-modal"


$(".show-modal").on("click", function (e) {
    e.preventDefault();
    var $this = $(this);
    var name;
    if ($this.attr('data-modal')) {
        name = "#" + $this.data("modal");
    } else {
        name = $this.attr('href')
    }
    $(name)
        .modal({
            transition: "fade",
            closable: false,
            autofocus: false,
            detachable: false,
        })
        .modal('show')
});
